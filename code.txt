
    public double calculateGrade() {
        //TODO 17: calculate gradeScore as per the instructions above
        double assignmentScore;
        double quizScore;

        if (course.subject.title.contains("Online")) {
            assignmentScore = (double) course.assignmentMarks / 30 * 10;
            quizScore = (double) course.quizMarks / 10 * 10;
        } else {
            assignmentScore = (double) course.assignmentMarks / 100 * 10;
            quizScore = (double) course.quizMarks / 30 * 10;
        }

        this.gradeScore = (assignmentScore + quizScore) / 2;
        return this.gradeScore;
    }
